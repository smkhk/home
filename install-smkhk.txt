#bin/bash
apt install git php -y

cat >start-home <<EOL

#bin/bash
mkdir ~/backup-home
cp ~/home/include/config.php ~/backup-home/config.php
cp -rf ~/home/voucher/. ~/backup-home/voucher/
cp -rf ~/home/img/. ~/backup-home/img/
rm -rf home
git clone https://github.com/smkhk/home.git
cp ~/backup-home/config.php ~/home/include/config.php

rm -rf ~/backup-home

EOL
cat >remove-home <<EOL
#bin/bash
rm -rf home
EOL
cat >.bashrc <<EOL	
~/start-home	
EOL
chmod +x start-home
